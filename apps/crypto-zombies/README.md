# keccak256

Hash function to map a byte input into a random 256-bit hexadecimal number. It takes a parameter of type bytes. We need to pack any parameters before calling it.

```js
keccak256(abi.encodePacked('aaaa'));
```

# Event

```js
// declare the event
event IntegersAdded(uint x, uint y, uint result);

function add(uint _x, uint _y) public returns (uint) {
  uint result = _x + _y;
  // fire an event to let the app know the function was called:
  emit IntegersAdded(_x, _y, result);
  return result;
}
```

Then, frontend can listen.

```js
YourContract.IntegersAdded(function (error, result) {
  // do something with result
});
```

# array.push()

Different number of components on the left hand side (1) than on the right hand side (0). Fix the error for this Solidity code `uint id = zombies.push(Zombie(_name, _dna)) - 1;`.

```js
zombies.push(Zombie(_name, _dna));
uint id = zombies.length - 1;
emit NewZombie(id, _name, _dna);
```

```js
// take the Zombie dna, and update our image
function generateZombie(id, name, dna) {
  let dnaStr = String(dna)
  // pad DNA with leading zeroes if it's less than 16 characters
  while (dnaStr.length < 16)
    dnaStr = "0" + dnaStr

  let zombieDetails = {
    // first 2 digits make up the head. We have 7 possible heads, so % 7
    // to get a number 0 - 6, then add 1 to make it 1 - 7. Then we have 7
    // image files named "head1.png" through "head7.png" we load based on
    // this number:
    headChoice: dnaStr.substring(0, 2) % 7 + 1,
    // 2nd 2 digits make up the eyes, 11 variations:
    eyeChoice: dnaStr.substring(2, 4) % 11 + 1,
    // 6 variations of shirts:
    shirtChoice: dnaStr.substring(4, 6) % 6 + 1,
    // last 6 digits control color. Updated using CSS filter: hue-rotate
    // which has 360 degrees:
    skinColorChoice: parseInt(dnaStr.substring(6, 8) / 100 * 360),
    eyeColorChoice: parseInt(dnaStr.substring(8, 10) / 100 * 360),
    clothesColorChoice: parseInt(dnaStr.substring(10, 12) / 100 * 360),
    zombieName: name,
    zombieDescription: "A Level 1 CryptoZombie",
  }
  return ZombieDetails
```
